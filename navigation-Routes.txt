MaterialApp(  
  title: 'Named Route Navigation',  

  initialRoute: '/',  
  routes: {   
    '/': (context) => HomeScreen(),   
    '/second': (context) => SecondScreen(),  
  },  
));  

onPressed: () {  
  Navigator.pushNamed(context, '/second');  
}, 

onPressed: () {  
  Navigator.pop(context);  
},  

onPressed: () {
  Navigator.push(context, MaterialPageRoute(builder: (context) => const NewPage()),);
},
